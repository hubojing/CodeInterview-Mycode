class Solution {
public:
    string replaceSpace(string s) {
        string str;
        int len = s.size();
        for(int i = 0; i < len; ++i)
        {
            if(s[i] == ' ')
            {
                str += "%20";
            }
            else
            {
                str += s[i];
            }
        }
        return str;
    }
};
